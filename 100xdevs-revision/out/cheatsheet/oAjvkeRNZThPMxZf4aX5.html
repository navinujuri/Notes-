Mongoose schemas (db.js)

Connect
const mongoose=require('mongoose');
mongoose.connect('mongodb+srv://.../paytm');
User schema
const userSchema=new mongoose.Schema({
  username:{type:String,required:true,unique:true,trim:true,lowercase:true,minLength:3,maxLength:30},
  password:{type:String,required:true,minLength:6},
  firstName:{type:String,required:true,trim:true,maxLength:50},
  lastName:{type:String,required:true,trim:true,maxLength:50}
});
unique:true builds an index; existing dupes must be cleared first.
Account schema (separate collection)
const accountSchema=new mongoose.Schema({
  userId:{type:mongoose.Schema.Types.ObjectId, ref:'User', required:true},
  balance:{type:Number, required:true}
});
Store money in its own collection, referenced by userId — not on User.
Export models
const User=mongoose.model('User', userSchema);
const Account=mongoose.model('Account', accountSchema);
module.exports={User, Account};

Auth routes + validation (routes/user.js)

zod input schema
const zod=require('zod');
const signupBody=zod.object({
  username:zod.string().email(),
  firstName:zod.string(), lastName:zod.string(),
  password:zod.string()
});
Signup → create user + account + JWT
router.post('/signup', async (req,res)=>{
  const {success}=signupBody.safeParse(req.body);
  if(!success) return res.status(411).json({message:'Incorrect inputs'});
  if(await User.findOne({username:req.body.username}))
    return res.status(411).json({message:'Email already taken'});
  const user=await User.create(req.body);
  await Account.create({userId:user._id, balance:1+Math.random()*10000});
  const token=jwt.sign({userId:user._id}, JWT_SECRET);
  res.json({message:'User created', token});
});
safeParse returns {success,data} — never throws. Give random starting balance.
Signin → verify + JWT
router.post('/signin', async (req,res)=>{
  const {success}=signinBody.safeParse(req.body);
  if(!success) return res.status(411).json({message:'Incorrect inputs'});
  const user=await User.findOne({username:req.body.username, password:req.body.password});
  if(user) return res.json({token: jwt.sign({userId:user._id}, JWT_SECRET)});
  res.status(411).json({message:'Error while logging in'});
});
jwt.sign({userId}) — put the id in the payload; read it back in middleware.
Update (protected)
router.put('/', authMiddleware, async (req,res)=>{
  await User.updateOne({_id:req.userId}, req.body);
  res.json({message:'Updated successfully'});
});
Bulk search (for send-money list)
router.get('/bulk', async (req,res)=>{
  const filter=req.query.filter||'';
  const users=await User.find({$or:[
    {firstName:{$regex:filter}}, {lastName:{$regex:filter}}
  ]});
  res.json({user: users.map(u=>({username:u.username,firstName:u.firstName,lastName:u.lastName,_id:u._id}))});
});

Middleware & JWT

config.js
module.exports={ JWT_SECRET:'your-super-secret' };
authMiddleware
const jwt=require('jsonwebtoken');
const {JWT_SECRET}=require('./config');
function authMiddleware(req,res,next){
  const h=req.headers.authorization;
  if(!h || !h.startsWith('Bearer ')) return res.status(403).json({});
  try{
    const {userId}=jwt.verify(h.split(' ')[1], JWT_SECRET);
    req.userId=userId;
    next();
  }catch(e){ return res.status(403).json({}); }
}
module.exports={authMiddleware};
Token comes as 'Bearer <token>' — split on space. Attach userId to req, then next().

Account & transactions (routes/account.js)

Get balance (protected)
router.get('/balance', authMiddleware, async (req,res)=>{
  const acc=await Account.findOne({userId:req.userId});
  res.json({balance: acc.balance});
});
Transfer with a DB transaction
const {default:mongoose}=require('mongoose');
router.post('/transfer', authMiddleware, async (req,res)=>{
  const session=await mongoose.startSession();
  session.startTransaction();
  const {amount,to}=req.body;
  const acc=await Account.findOne({userId:req.userId}).session(session);
  if(!acc || acc.balance<amount){ await session.abortTransaction(); return res.status(400).json({message:'Insufficient balance'}); }
  const toAcc=await Account.findOne({userId:to}).session(session);
  if(!toAcc){ await session.abortTransaction(); return res.status(400).json({message:'Invalid account'}); }
  await Account.updateOne({userId:req.userId},{$inc:{balance:-amount}}).session(session);
  await Account.updateOne({userId:to},   {$inc:{balance: amount}}).session(session);
  await session.commitTransaction();
  res.json({message:'Transfer successful'});
});
All reads/writes must use .session(session). Any abort rolls back both $inc — no half-transfers.
Why $inc, not read-then-set
{$inc:{balance:-amount}}  // atomic
read → subtract → save races under concurrency; $inc is a single atomic op.

Quick tips & gotchas

  • req.body is undefined unless app.use(express.json()) runs before your routers — and app.use(cors()) is required or the React app's fetches get blocked.
  • zod safeParse returns {success,data} and never throws; check !success and return 411 before touching the DB.
  • Keep money in a separate Account collection keyed by userId — don't put balance on the User doc.
  • Transfers MUST use mongoose.startSession()/startTransaction() with .session() on every op; abortTransaction on any failure so you never debit without crediting.
  • Use $inc for balance changes — read-modify-write loses updates under concurrent transfers.
  • JWT flow: sign {userId} on signup/signin, send as 'Bearer <token>', verify + set req.userId in authMiddleware. The 'Bearer ' prefix must match on both ends.
  • Frontend keeps the token in localStorage and attaches Authorization headers manually; there are no cookies in this project.
  • MongoDB transactions require a replica set (Atlas works by default; a bare local mongod does not) — otherwise startTransaction throws.
  • unique:true on username only enforces after the index builds; wipe duplicate docs before relying on it, and still do a findOne pre-check for a clean 411.
